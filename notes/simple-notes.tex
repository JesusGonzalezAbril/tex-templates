%     - Automatically increments lecture number.
%

\documentclass{report}

\usepackage[a4paper,top=3cm,bottom=3cm,left=3cm,right=3cm,marginparwidth=2cm]{geometry} % Page dimensions

\usepackage[english]{babel}
\usepackage{amsmath} 
\usepackage{amssymb}
\usepackage{amsthm}
\usepackage{xcolor}

%% Theorem macros %%
\theoremstyle{definition}

\newtheorem{definition}{Definition}[section] % Numbered within section
\newtheorem{theorem}[definition]{Theorem}
\newtheorem{lemma}[definition]{Lemma} 
\newtheorem{corollary}[definition]{Corollary}

%% Lecture counter %%
\newcounter{lecture}
\newcommand{\lec}[1]{
    \stepcounter{lecture}
    \subsection*{\color{orange} Lecture \thelecture\ - #1}
}
